# 获取每个部门当前员工最高薪资信息

## 问题描述
获取每个部门中当前员工薪水最高的相关信息，包括部门编号，员工编号和薪资
## 原始代码
```` sql
select
from
    dept_emp d
    join salaries s on d.emp_no =s.emp_no
    join(
        select
            d1.dept_no dept_no, 
            max(salary) as max_salary
        from
            dept_emp d1
            join salaries s1 on d1.emp_no =s1.emp_no
        group by
            d1.dept_no
    ) m on d.dept_no=m.dept_no && s.salary=m.salary
where
    d.to_date='9999-01-01' && s.todate='9999-01-01'
order by
    d.dept_no
